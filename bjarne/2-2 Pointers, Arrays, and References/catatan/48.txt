u8R"(folder\file)"  // UTF-8 raw string
u"folder\\file"     // UTF-16 string
uR"(folder\file)"   //UTF-16 raw string
U"folder\\file"     // UTF-32 string
UR"(folder\file)"   //UTF-32 raw string
